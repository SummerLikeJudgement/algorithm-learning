# 哈希
## 1.两数之和
简单题
1. 【排序+双指针】时间复杂度$O(NlogN)$。利用pair存储原数组的值和下标，对值排序后利用双指针找到答案
```cpp
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        vector<pair<int, int>> all;
        int cnt = 0;
        for(auto t=nums.begin() ; t!=nums.end() ; t++, cnt++)
        {
            all.push_back({*t, cnt});
        }
        sort(all.begin(), all.end());
        int i=0 , j=cnt-1;
        while(all[i].first+all[j].first != target)
        {
            if(all[i].first+all[j].first > target)
                j--;
            else if(all[i].first+all[j].first < target)
                i++;
        }
        return {all[i].second, all[j].second};
    }
};
```
2. 【哈希表】时间复杂度$O(N)$。对暴力解法的优化，使用hash表优化查询target-x的过程
```cpp
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int, int> h; // 记录遍历过的值-位置
        for(int i=0 ; i<nums.size() ; i++)
        {
            int x = nums[i];
            if(h.find(target - x) != h.end())
                return {i, h[target-x]};
            else
                h[x] = i;
        }
        return {}; // 一定要返回空，否则编译不通过
    }
};
```

## 49.字母异位词分组
中等题
1. 【排序+哈希】时间复杂度$O(NKlogK)$。遍历所有的string，将其排序后存入hash表
```cpp
class Solution {
public:
    vector<vector<string>> groupAnagrams(vector<string>& strs) {
        vector<vector<string>> ans;
        int cnt = 0;
        unordered_map<string, int> h;
        // 
        for(auto t=strs.begin() ; t!=strs.end() ; t++)
        {
            string temp = *t;
            sort(temp.begin(), temp.end());
            if(h.find(temp) != h.end())
            {
                int pos = h[temp];
                ans[pos].push_back(*t);
            }
            else
            {
                int pos = cnt++;
                h[temp] = pos;
                ans.push_back({});
                ans[pos].push_back(*t);
            }
        }

        return ans;
    }
};
```

2. 【计数+哈希】$O(n(k+|\sum|))$。时间复杂度$$遍历所有string，对string内的所有字母计数
```cpp
class Solution {
public:
    vector<vector<string>> groupAnagrams(vector<string>& strs) {
        // 自定义对 array<int, 26> 类型的哈希函数
        auto arrayHash = [fn = hash<int>{}] (const array<int, 26>& arr) -> size_t {
            return accumulate(arr.begin(), arr.end(), 0u, [&](size_t acc, int num) {
                return (acc << 1) ^ fn(num);
            });
        };

        unordered_map<array<int, 26>, vector<string>, decltype(arrayHash)> mp(0, arrayHash);
        for (string& str: strs) {
            array<int, 26> counts{};
            int length = str.length();
            for (int i = 0; i < length; ++i) {
                counts[str[i] - 'a'] ++;
            }
            mp[counts].emplace_back(str);
        }
        vector<vector<string>> ans;
        for (auto it = mp.begin(); it != mp.end(); ++it) {
            ans.emplace_back(it->second);
        }
        return ans;
    }
};
```

# 多维动态规划
## 1143.最长公共子序列
中等题
【动态规划】时间复杂度$O(mn)$。动态规划的经典题目
```cpp
class Solution {
public:
    int longestCommonSubsequence(string text1, string text2) {
        const int N = 1010;
        int f[N][N];
        for(int i=1 ; i<=text1.size(); i++)
            for(int j=1 ; j<=text2.size() ; j++)
            {
                if(text1[i-1] == text2[j-1])
                    f[i][j] = f[i-1][j-1] + 1;
                else
                    f[i][j] = max(f[i][j-1], f[i-1][j]);
            }
        return f[text1.size()][text2.size()];
    }
};
```
## 64.最小路径和
中等题
1. 【动态规划】时间复杂度$O(MN)$。二维，f(i,j)表示从起点到(i,j)的所有路径的数字和的最小值；f(i,j) = min(f(i-1,j), f(i,j-1))+w(i,j)
```cpp
class Solution {
public:
    int minPathSum(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        const int N = 205;
        int f[N][N];
        // init
        memset(f, 0x3f, sizeof f);
        f[1][1] = grid[0][0];

        for(int i=1 ; i<=m ; i++)
            for(int j=1 ; j<=n ; j++)
            {
                if(i == 1 && j == 1)
                    continue;
                f[i][j] = min(f[i-1][j], f[i][j-1]) + grid[i-1][j-1];
            }

        return f[m][n];
    }
};
```
2. 【动态规划】时间复杂度$O(MN)$。一维（优化空间复杂度），f(j)表示从起点到i行j列的所有路径的数字和的最小值；f(j) = min(f(j), f(j-1))+w(i,j)且j从小到大（使用的是更新后的f(j-1)值）
```cpp

class Solution {
public:
    int minPathSum(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        const int N = 205;
        int f[N];
        // init
        memset(f, 0x3f, sizeof f);

        for(int i=1 ; i<=m ; i++)
            for(int j=1 ; j<=n ; j++) // j仍然从小到大，使用的是更新后的j-1值
                if(i == 1 && j == 1)
                    f[j] = grid[0][0];
                else
                    f[j] = min(f[j-1], f[j]) + grid[i-1][j-1];

        return f[n];
    }
};
```
# 回溯
## 79. 单词搜索
【dfs】时间复杂度$O(MN⋅3^L)$。典型dfs问题，重点在于记录+还原的回溯以及边界条件
```cpp
class Solution {
public:
    int dx[4] = {0, 0, -1, 1};
    int dy[4] = {1, -1, 0, 0};

    bool dfs(vector<vector<char>>& board, string& word, int x, int y, int cnt)
    {
        if(cnt == word.size())
            return true;
        for(int k=0 ; k<4 ; k++)
        {
            int tx = x + dx[k];
            int ty = y + dy[k];
            if(tx<0 || tx>=board.size() || ty<0 || ty>=board[0].size())
                continue;
            if(board[tx][ty] == word[cnt])
            {
                char temp = board[tx][ty];
                board[tx][ty] = '0';
                if(dfs(board, word, tx, ty, cnt+1))
                    return true;
                board[tx][ty] = temp;
            }

        }
        return false;
    }


    bool exist(vector<vector<char>>& board, string word) {
        int m = board.size();
        int n = board[0].size();


        for(int i=0 ; i<m ; i++)
            for(int j=0 ; j<n ; j++)
            {
                if(board[i][j] == word[0])
                {
                    char temp = board[i][j];
                    board[i][j] = '0';
                    if(dfs(board, word, i, j, 1))
                        return true;
                    board[i][j] = temp;
                }
            }
        return false;
                
    }
};
```
# 堆
## 215. 数组中的第K个最大元素
1. 【大根堆】时间复杂度O$(NlogN)$
将数组构造大根堆后pop出k-1个元素即可
```cpp
class Solution {
public:
    static const int N = 1e5+10;
    int h[N], size_; // 大根堆
    // 尝试将x位置的值向下到正确的位置
    void down(int x)
    {
        int t = x;
        if(2*x <= size_ && h[2*x] > h[t]) t = 2*x;
        if(2*x+1 <= size_ && h[2*x+1] > h[t]) t = 2*x+1;

        if(t!=x)
        {
            swap(h[x], h[t]);
            down(t);
        }
    }
    // 尝试将x位置的值向上到正确位置
    void up(int x)
    {
        while(x/2 && h[x]>h[x/2])
        {
            swap(h[x], h[x/2]);
            x /= 2;
        }
    }
    int findKthLargest(vector<int>& nums, int k) {
        // 初始化大根堆
        size_ = nums.size();
        for(int i=1 ; i<=size_ ; i++)
            h[i] = nums[i-1];
        for(int i=size_/2 ; i ; i--)
            down(i);
        while(--k)
        {
            swap(h[1], h[size_]);
            size_--;
            down(1);
        }
        return h[1];
    }
};
```
2. 【快速选择】时间复杂度理论$O(N)$
优化快排找到下标元素的方法
```cpp
class Solution {
public:
    // 在q的[l,r]中找到最终下标为k的数
    int quick_select(int q[], int l, int r, int k)
    {
        if(l == r) return q[l]; 
        int i = l-1, j = r+1;
        int base = q[l];

        while(i<j)
        {
            while(q[++i] < base){}
            while(q[--j] > base){}

            if(i<j)
                swap(q[i], q[j]);
        }

        if(k <= j) 
            return quick_select(q, l, j, k);
        else 
            return quick_select(q, j+1, r, k);
    }
    int findKthLargest(vector<int>& nums, int k) {
        int q[100010];
        int cnt = nums.size();
        for(int i=0 ; i<cnt ; i++)
            q[i] = nums[i];

        int ans = quick_select(q, 0, cnt-1, cnt-k);

        return ans;
    }
};
```
# 图论
## 200.岛屿数量
中等题
1. 【DFS】时间复杂度$O(MN)$。从陆地方格dfs所有邻近的陆地方格，遍历后置为海洋
```cpp
class Solution {
public:
    // bias
    int dx[4] = {0, 0, -1, 1};
    int dy[4] = {1, -1, 0, 0};
    // (x,y)'s DFS
    void dfs(int x, int y, vector<vector<char>>& grid)
    {
        grid[x][y] = '0';

        for(int i=0 ; i<4 ; i++)
        {
            int tx = x + dx[i];
            int ty = y + dy[i];
            if(tx<0 || tx>=grid.size())
                continue;
            if(ty<0 || ty>=grid[tx].size())
                continue;
            if(grid[tx][ty] == '1')
            {
                dfs(tx, ty, grid);
            }
        }
    }

    int numIslands(vector<vector<char>>& grid) {
        int ans = 0;
        
        for(int i=0 ; i<grid.size() ; i++)
            for(int j=0 ; j<grid[i].size() ; j++)
                if(grid[i][j] == '1' )
                {
                    dfs(i, j, grid);
                    ans++;
                }

        return ans;
    }
};
```

# 矩阵

## 54.螺旋矩阵
中等题
1. 【模拟】时间复杂度$O(MN)$。将矩阵看作若干圈，从外向内输出元素
```cpp
class Solution {
public:
    vector<int> spiralOrder(vector<vector<int>>& matrix) {
        int m = matrix.size();
        int n = matrix[0].size();
        vector<int> ans;
        
        int i = 0, j = 0;

        while(m>1 && n>1)
        {
            ans.push_back(matrix[i][j]);
            for(int k=0 ; k<n-1 ; k++)
            {
                j++;
                ans.push_back(matrix[i][j]);
            }
            for(int k=0 ; k<m-1 ; k++)
            {
                i++;
                ans.push_back(matrix[i][j]);
            }
            for(int k=0 ; k<n-1 ; k++)
            {
                j--;
                ans.push_back(matrix[i][j]);
            }
            for(int k=0 ; k<m-2 ; k++)
            {
                i--;
                ans.push_back(matrix[i][j]);
            }
            j++;
            m -= 2;
            n -= 2;
        }
        if(m == 0 || n==0)
            return ans;
        if(m == 1)
        {
            ans.push_back(matrix[i][j]);
            for(int k=0 ; k<n-1 ; k++)
            {
                j++;
                ans.push_back(matrix[i][j]);
            }
        }
        else if(n == 1)
        {
            ans.push_back(matrix[i][j]);
            for(int k=0 ; k<m-1 ; k++)
            {
                i++;
                ans.push_back(matrix[i][j]);
            }
        }
        

        return ans;
    }
};
```

2. 【模拟】时间复杂度$O(MN)$。将矩阵看作若干层，每层遍历结束后直接将边界移动
```cpp
class Solution {
public:
    vector<int> spiralOrder(vector<vector<int>>& matrix) {
        vector <int> ans;
        if(matrix.empty()) return ans; //若数组为空，直接返回答案
        int u = 0; //赋值上下左右边界
        int d = matrix.size() - 1;
        int l = 0;
        int r = matrix[0].size() - 1;
        while(true)
        {
            for(int i = l; i <= r; ++i) ans.push_back(matrix[u][i]); //向右移动直到最右
            if(++ u > d) break; //重新设定上边界，若上边界大于下边界，则遍历遍历完成，下同
            for(int i = u; i <= d; ++i) ans.push_back(matrix[i][r]); //向下
            if(-- r < l) break; //重新设定有边界
            for(int i = r; i >= l; --i) ans.push_back(matrix[d][i]); //向左
            if(-- d < u) break; //重新设定下边界
            for(int i = d; i >= u; --i) ans.push_back(matrix[i][l]); //向上
            if(++ l > r) break; //重新设定左边界
        }
        return ans;
    }
};
```

## 136.只出现一次的数字
简单题
【位运算】时间复杂度$O(N)$。相同数字异或结果为0，所有数字异或结果就只剩下单独的元素了
```cpp
class Solution {
public:
    int singleNumber(vector<int>& nums) {
        int ans = nums[0];
        for(int i=1 ; i<nums.size() ; i++)
            ans = ans ^ nums[i];

        return ans;
    }
};
```
