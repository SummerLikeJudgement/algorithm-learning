# 哈希
## 1.两数之和
简单题
1. 【排序+双指针】时间复杂度$O(NlogN)$
利用pair存储原数组的值和下标，对值排序后利用双指针找到答案
```cpp
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        vector<pair<int, int>> all;
        int cnt = 0;
        for(auto t=nums.begin() ; t!=nums.end() ; t++, cnt++)
        {
            all.push_back({*t, cnt});
        }
        sort(all.begin(), all.end());
        int i=0 , j=cnt-1;
        while(all[i].first+all[j].first != target)
        {
            if(all[i].first+all[j].first > target)
                j--;
            else if(all[i].first+all[j].first < target)
                i++;
        }
        return {all[i].second, all[j].second};
    }
};
```
2. 【哈希表】时间复杂度$O(N)$
对暴力解法的优化，使用hash表优化查询target-x的过程
```cpp
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int, int> h; // 记录遍历过的值-位置
        for(int i=0 ; i<nums.size() ; i++)
        {
            int x = nums[i];
            if(h.find(target - x) != h.end())
                return {i, h[target-x]};
            else
                h[x] = i;
        }
        return {}; // 一定要返回空，否则编译不通过
    }
};
```
# 多维动态规划
## 1143.最长公共子序列
【动态规划】时间复杂度$O(mn)$
动态规划的经典题目
```cpp
class Solution {
public:
    int longestCommonSubsequence(string text1, string text2) {
        const int N = 1010;
        int f[N][N];
        for(int i=1 ; i<=text1.size(); i++)
            for(int j=1 ; j<=text2.size() ; j++)
            {
                if(text1[i-1] == text2[j-1])
                    f[i][j] = f[i-1][j-1] + 1;
                else
                    f[i][j] = max(f[i][j-1], f[i-1][j]);
            }
        return f[text1.size()][text2.size()];
    }
};
```
# 堆
## 215. 数组中的第K个最大元素
1. 【大根堆】时间复杂度O$(NlogN)$
将数组构造大根堆后pop出k-1个元素即可
```cpp
class Solution {
public:
    static const int N = 1e5+10;
    int h[N], size_; // 大根堆
    // 尝试将x位置的值向下到正确的位置
    void down(int x)
    {
        int t = x;
        if(2*x <= size_ && h[2*x] > h[t]) t = 2*x;
        if(2*x+1 <= size_ && h[2*x+1] > h[t]) t = 2*x+1;

        if(t!=x)
        {
            swap(h[x], h[t]);
            down(t);
        }
    }
    // 尝试将x位置的值向上到正确位置
    void up(int x)
    {
        while(x/2 && h[x]>h[x/2])
        {
            swap(h[x], h[x/2]);
            x /= 2;
        }
    }
    int findKthLargest(vector<int>& nums, int k) {
        // 初始化大根堆
        size_ = nums.size();
        for(int i=1 ; i<=size_ ; i++)
            h[i] = nums[i-1];
        for(int i=size_/2 ; i ; i--)
            down(i);
        while(--k)
        {
            swap(h[1], h[size_]);
            size_--;
            down(1);
        }
        return h[1];
    }
};
```
2. 【快速选择】时间复杂度理论$O(N)$
优化快排找到下标元素的方法
```cpp
class Solution {
public:
    // 在q的[l,r]中找到最终下标为k的数
    int quick_select(int q[], int l, int r, int k)
    {
        if(l == r) return q[l]; 
        int i = l-1, j = r+1;
        int base = q[l];

        while(i<j)
        {
            while(q[++i] < base){}
            while(q[--j] > base){}

            if(i<j)
                swap(q[i], q[j]);
        }

        if(k <= j) 
            return quick_select(q, l, j, k);
        else 
            return quick_select(q, j+1, r, k);
    }
    int findKthLargest(vector<int>& nums, int k) {
        int q[100010];
        int cnt = nums.size();
        for(int i=0 ; i<cnt ; i++)
            q[i] = nums[i];

        int ans = quick_select(q, 0, cnt-1, cnt-k);

        return ans;
    }
};
```
